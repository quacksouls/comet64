// assume that each of the registers "int" and "reg" is
// initially set to zero.

check input = null;   // Whether we run out of numbers.
jump if true: print;  // If true, branch to "print".
jump if false: end;   // Otherwise, branch to "end".
print:                // The "print" branch.
    reg = reg / int;  // The arithmetic mean.
