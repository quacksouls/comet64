> Task: Make both grid patterns identical.  First grid can be written by
> outputting true or false.  Second grid shows the solution.
// We want to print this pattern:
//
// xo
// xoo
// xooo
// xoooo
// xooooo
// xoooooo
// xooooooo
//
// Use the register "reg" to keep track of the number of "o" to print.
// The register "int" can be used to keep track of the number of "o"
// we have printed so far.
loop:
    check int < reg;
    jump if false: end;
    output = false;
    int++;
    jump to: loop;
end:
    output = true;
    reg++;
    int = 0;
